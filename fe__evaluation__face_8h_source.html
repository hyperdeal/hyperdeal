133 void
134 reinit(typename MatrixFree<dim_x, dim_v, Number, VectorizedArrayType>::ID
135 face_index) override
136 {
137 PARENT::reinit(face_index);
138 this->type = face_index.type;
139 this->is_ecl = false;
140
141 if (this->type == SpaceType::X)
142 {
143 phi_face_x.reinit(this->macro_cell_x);
144 phi_v.reinit(this->macro_cell_v);
145 }
146 else
147 {
148 phi_x.reinit(this->macro_cell_x);
149 phi_face_v.reinit(this->macro_cell_v);
150 }
151
152 if (this->type == SpaceType::X)
153 {
154 this->n_filled_lanes =
155 this->matrix_free_x.n_active_entries_per_face_batch(
156 this->macro_cell_x);
157 }
158 else
159 {
160 this->n_filled_lanes =
161 this->matrix_free_x.n_active_entries_per_cell_batch(
162 this->macro_cell_x);
163 }
164
165 // get direction of face
166 if (type == SpaceType::X)
167 {
168 const dealii::internal::MatrixFreeFunctions::FaceToCellTopology<
169 n_vectors> &faces =
170 this->matrix_free_x.get_face_info(this->macro_cell_x);
171 this->face_no = this->is_minus_face ? faces.interior_face_no :
172 faces.exterior_face_no;
173 }
174 else
175 {
176 const dealii::internal::MatrixFreeFunctions::FaceToCellTopology<
177 n_vectors_v> &faces =
178 this->matrix_free_v.get_face_info(this->macro_cell_v);
179 this->face_no = this->is_minus_face ? faces.interior_face_no :
180 faces.exterior_face_no;
181 this->face_no += dim_x * 2;
182 }
183 }

190 void
191 reinit(typename MatrixFree<dim_x, dim_v, Number, VectorizedArrayType>::ID
192 cell_index,
193 unsigned int face)
194 {
195 PARENT::reinit(cell_index);
196 this->type = (face >= 2 * dim_x) ? SpaceType::V : SpaceType::X;
197 this->is_ecl = true;
198
199 // delegate reinit to deal.II data structures
200 if (this->type == SpaceType::X)
201 {
202 phi_face_x.reinit(this->macro_cell_x, face);
203 phi_v.reinit(this->macro_cell_v);
204 }
205 else
206 {
207 phi_x.reinit(this->macro_cell_x);
208 phi_face_v.reinit(this->macro_cell_v, face - 2 * dim_x);
209 }
210
211 // how many lanes are filled?
212 {
213 this->n_filled_lanes =
214 this->matrix_free_x.n_active_entries_per_cell_batch(
215 this->macro_cell_x);
216 }
217
218 // get direction of face
219 {
220 this->face_no = face;
221 }

240 void
241 read_dof_values(const dealii::LinearAlgebra::SharedMPI::Vector<Number> &src,
242 VectorizedArrayType *data) const
243 {
244 if (this->matrix_free.are_ghost_faces_supported())
245 {
246 // for comments see dealii::FEEvaluation::reinit
247 const unsigned int face_orientation =
248 is_ecl ? 0 :
249 ((this->is_minus_face ==
250 (this->matrix_free.get_face_info()
251 .face_orientations[0][this->macro] >= 8)) ?
252 (this->matrix_free.get_face_info()
253 .face_orientations[0][this->macro] %
254 8) :
255 0);
256
257 internal::MatrixFreeFunctions::ReadWriteOperation<Number>(
258 this->matrix_free.get_dof_info(),
259 this->matrix_free.get_face_info(),
260 this->matrix_free.get_shape_info())
261 .template process_face<dim_x, dim_v, degree>(
262 internal::MatrixFreeFunctions::
263 VectorReader<Number, VectorizedArrayType>(),
264 src.shared_vector_data(),
265 data,
266 (is_ecl == false || this->is_minus_face) ?
267 &this->face_no :
268 &this->matrix_free.get_face_info()
269 .no_faces[3][(2 * dim * this->macro + this->face_no) *
270 n_vectors],
271 (is_ecl == false || this->is_minus_face) ?
272 &face_orientation :
273 &this->matrix_free.get_face_info().face_orientations
274 [3][(2 * dim * this->macro + this->face_no) * n_vectors],
275 this->type == SpaceType::X ? 0 : 8,
276 this->macro,
277 is_ecl ? 2 : !is_minus_face,
278 (is_ecl == false || this->is_minus_face) ?
279 this->macro :
280 2 * dim * this->macro + this->face_no,
281 !is_minus_face + (is_ecl ? 2 : 0));
282 }
283 else
284 {
285 AssertThrow(false, dealii::StandardExceptions::ExcNotImplemented());
286 }

381 inline DEAL_II_ALWAYS_INLINE //
382 void
383 submit_value(VectorizedArrayType *__restrict data_ptr,
384 const VectorizedArrayType &value,
385 const unsigned int q,
386 const unsigned int qx,
387 const unsigned int qv) const
388 {
389 Assert(stype == this->type, dealii::ExcMessage("Types do not match!"));
390
391 // note: this if-statement is evaluated at compile time
392 if (stype == SpaceType::X)
393 data_ptr[q] = -value * phi_face_x.JxW(qx) *
394 phi_v.JxW(qv)[n_vectors_v == 1 ? 0 : this->lane_y];
395 else if (stype == SpaceType::V)
396 data_ptr[q] = -value * phi_x.JxW(qx) *
397 phi_face_v.JxW(qv)[n_vectors_v == 1 ? 0 : this->lane_y];
398 }

471 inline dealii::Point<DIM, VectorizedArrayType>
472 get_quadrature_point(const unsigned int q) const
473 {
474 dealii::Point<DIM, VectorizedArrayType> temp;
475
476 if (type == SpaceType::X)
477 {
478 const auto t1 =
479 this->phi_face_x.quadrature_point(q % N_Q_POINTS_1_FACE);
480 for (int i = 0; i < dim_x; i++)
481 temp[i] = t1[i];
482 const auto t2 = this->phi_v.quadrature_point(q / N_Q_POINTS_1_FACE);
483 for (int i = 0; i < dim_v; i++)
484 temp[i + dim_x] = t2[i][n_vectors_v == 1 ? 0 : this->lane_y];
485 }
486 else
487 {
488 const auto t1 = this->phi_x.quadrature_point(q % N_Q_POINTS_1_CELL);
489 for (int i = 0; i < dim_x; i++)
490 temp[i] = t1[i];
491 const auto t2 =
492 this->phi_face_v.quadrature_point(q / N_Q_POINTS_1_CELL);
493 for (int i = 0; i < dim_v; i++)
494 temp[i + dim_x] = t2[i][n_vectors_v == 1 ? 0 : this->lane_y];
495 }
496
497 return temp;
498 }
