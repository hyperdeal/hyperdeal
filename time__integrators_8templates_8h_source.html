91 template <typename Number, typename VectorType>
92 void
93 LowStorageRungeKuttaIntegrator<Number, VectorType>::perform_time_step(
94 VectorType & solution,
95 const Number &current_time,
96 const Number &time_step,
97 const std::function<void(const VectorType &, VectorType &, const Number)>
98 &op)
99 {
100 const auto &local_elements = solution.locally_owned_elements();
101
102 // definition of a stage
103 auto perform_stage = [&](const Number current_time,
104 const Number factor_solution,
105 const Number factor_ai,
106 const VectorType &current_Ti,
107 VectorType & vec_Ki,
108 VectorType & solution,
109 VectorType & next_Ti) {
110 // call operator
111 op(current_Ti, vec_Ki, current_time);
112
113 const Number ai = factor_ai;
114 const Number bi = factor_solution;
115 if (ai == Number())
116 {
117 for (const auto i : local_elements)
118 {
119 const Number K_i = vec_Ki(i);
120 const Number sol_i = solution(i);
121 solution(i) = sol_i + bi * K_i;
122 }
123 }
124 else
125 {
126 for (const auto i : local_elements)
127 {
128 const Number K_i = vec_Ki(i);
129 const Number sol_i = solution(i);
130 solution(i) = sol_i + bi * K_i;
131 next_Ti(i) = sol_i + ai * K_i;
132 }
133 }
134 };
135
136
137 // perform first stage
138 if (only_Ti_is_ghosted)
139 {
140 // swap solution and Ti
141 for (const auto i : local_elements)
142 vec_Ti(i) = solution(i);
143
144 perform_stage(current_time,
145 bi[0] * time_step,
146 ai[0] * time_step,
147 vec_Ti,
148 vec_Ki,
149 solution,
150 vec_Ti);
151 }
152 else
153 {
154 perform_stage(current_time,
155 bi[0] * time_step,
156 ai[0] * time_step,
157 solution,
158 vec_Ti,
159 solution,
160 vec_Ti);
161 }
162
163
164 // perform rest stages
165 Number sum_previous_bi = 0;
166 for (unsigned int stage = 1; stage < bi.size(); ++stage)
167 {
168 const Number c_i = sum_previous_bi + ai[stage - 1];
169 perform_stage(current_time + c_i * time_step,
170 bi[stage] * time_step,
171 (stage == bi.size() - 1 ? 0 : ai[stage] * time_step),
172 vec_Ti,
173 vec_Ki,
174 solution,
175 vec_Ti);
176 sum_previous_bi += bi[stage - 1];
177 }
178 }

181 unsigned int
182 LowStorageRungeKuttaIntegrator<Number, VectorType>::n_stages() const
183 {
184 return bi.size();
185 }
