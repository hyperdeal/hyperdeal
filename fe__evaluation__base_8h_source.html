150 template <int dim_x,
151 int dim_v,
152 int degree,
153 int n_points,
154 typename Number,
155 typename VectorizedArrayType>
156 void
157 FEEvaluationBase<dim_x,
158 dim_v,
159 degree,
160 n_points,
161 Number,
162 VectorizedArrayType>::
163 reinit(typename MatrixFree<dim_x, dim_v, Number, VectorizedArrayType>::ID
164 cell_index)
165 {
166 this->macro_cell_x = cell_index.x;
167 this->macro_cell_v = cell_index.v / n_vectors_v;
168 this->lane_y = cell_index.v % n_vectors_v;
169 this->macro = cell_index.macro;
170
171 Assert(this->lane_y < this->n_vectors_v,
172 dealii::ExcIndexRange(this->lane_y, 0, n_vectors_v));
173 }
