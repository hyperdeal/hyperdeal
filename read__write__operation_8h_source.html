130  template <typename Number>
131  template <int dim,
132  int degree,
133  typename VectorOperation,
134  typename VectorizedArrayType>
135  void
136  ReadWriteOperation<Number>::process_cell(
137  const VectorOperation & operation,
138  const std::vector<dealii::ArrayView<const Number>> &global,
139  VectorizedArrayType * local,
140  const unsigned int cell_batch_number) const
141  {
142  static const unsigned int v_len = VectorizedArrayType::size();
143  static const unsigned int n_dofs_per_cell =
144  dealii::Utilities::pow<unsigned int>(degree + 1, dim);
145 
146  // step 1: get pointer to the first dof of the cell in the sm-domain
147  std::array<Number *, v_len> global_ptr;
148  std::fill(global_ptr.begin(), global_ptr.end(), nullptr);
149 
150  for (unsigned int v = 0;
151  v < n_vectorization_lanes_filled[2][cell_batch_number] &&
152  v < v_len;
153  v++)
154  {
155  const auto sm_ptr =
156  dof_indices_contiguous_ptr[2][v_len * cell_batch_number + v];
157  global_ptr[v] =
158  const_cast<Number *>(global[sm_ptr.first].data()) + sm_ptr.second;
159  }
160 
161  // step 2: process dofs
162  if (n_vectorization_lanes_filled[2][cell_batch_number] == v_len)
163  // case 1: all lanes are filled -> use optimized function
164  operation.process_dofs_vectorized_transpose(n_dofs_per_cell,
165  global_ptr,
166  local);
167  else
168  // case 2: some lanes are empty
169  for (unsigned int i = 0; i < n_dofs_per_cell; ++i)
170  for (unsigned int v = 0;
171  v < n_vectorization_lanes_filled[2][cell_batch_number] &&
172  v < v_len;
173  v++)
174  operation.process_dof(global_ptr[v][i], local[i][v]);
175  }

193  const unsigned int cell_batch_number,

195  const unsigned int face_batch_number,

198  static const unsigned int dim = dim_x + dim_v;
199  static const unsigned int v_len = VectorizedArrayType::size();
200  static const unsigned int n_dofs_per_face =
201  dealii::Utilities::pow<unsigned int>(degree + 1, dim - 1);
202 
203  std::array<Number *, v_len> global_ptr;
204  std::fill(global_ptr.begin(), global_ptr.end(), nullptr);
205 
206  for (unsigned int v = 0;
207  v < n_vectorization_lanes_filled[cell_side][cell_batch_number] &&
208  v < v_len;
209  v++)
210  {
211  const auto sm_ptr =
212  dof_indices_contiguous_ptr[face_side]
213  [v_len * face_batch_number + v];
214  global_ptr[v] =
215  const_cast<Number *>(global[sm_ptr.first].data()) + sm_ptr.second;
216  }
217 
218  if (n_vectorization_lanes_filled[cell_side][cell_batch_number] ==
219  v_len &&
220  (face_side == 2 || face_all[face_side][face_batch_number]))
221  {
222  const auto &face_orientations_ =
223  face_orientations[face_orientation[0] + face_orientation_offset];
224 
225  if (face_side != 2 &&
226  face_type[face_side][v_len * face_batch_number])
227  {
228  // case 1: read from buffers
229  for (unsigned int i = 0; i < n_dofs_per_face; ++i)
230  {
231  const unsigned int i_ = (((dim_x <= 2) && (dim_v <= 2)) ||
232  face_orientation[0] == 0) ?
233  i :
234  face_orientations_[i];
235 
236  for (unsigned int v = 0; v < v_len; ++v)
237  operation.process_dof(global_ptr[v][i], local[i_][v]);
238  }
239  }
240  else
241  {
242  // case 2: read from shared memory
243  for (unsigned int i = 0; i < n_dofs_per_face; ++i)
244  {
245  const unsigned int i_ = (((dim_x <= 2) && (dim_v <= 2)) ||
246  face_orientation[0] == 0) ?
247  i :
248  face_orientations_[i];
249 
250  for (unsigned int v = 0; v < v_len; ++v)
251  operation.process_dof(
252  global_ptr[v][face_to_cell_index_nodal[face_no[0]][i]],
253  local[i_][v]);
254  }
255  }
256  }
257  else
258  for (unsigned int v = 0;
259  v < n_vectorization_lanes_filled[cell_side][cell_batch_number] &&
260  v < v_len;
261  v++)
262  {
263  const auto &face_orientations_ =
264  face_orientations[face_orientation[face_side == 3 ? v : 0] +
265  face_orientation_offset];
266 
267  if (face_side != 2 &&
268  face_type[face_side][v_len * face_batch_number + v])
269  {
270  // case 1: read from buffers
271  for (unsigned int i = 0; i < n_dofs_per_face; ++i)
272  {
273  const unsigned int i_ =
274  (((dim_x <= 2) && (dim_v <= 2)) ||
275  face_orientation[face_side == 3 ? v : 0] == 0) ?
276  i :
277  face_orientations_[i];
278 
279  operation.process_dof(global_ptr[v][i], local[i_][v]);
280  }
281  }

283  {
284  // case 2: read from shared memory
285  for (unsigned int i = 0; i < n_dofs_per_face; ++i)
286  {

288  (((dim_x <= 2) && (dim_v <= 2)) ||
289  face_orientation[face_side == 3 ? v : 0] == 0) ?

296  local[i_][v]);
